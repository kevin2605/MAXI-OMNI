CREATE TABLE product(
	ProductID varchar(100) NOT NULL,
	ProductName TINYTEXT NOT NULL,
	Description TINYTEXT NOT NULL,
	ModalPrice FLOAT NOT NULL,
	SellingPrice FLOAT NOT NULL,
	isParent BOOLEAN DEFAULT FALSE,
	ChildID TINYTEXT DEFAULT NULL,
	StockValue int(6) NOT NULL,
	MainStock int(6) NOT NULL,
	ReserveStock int(6) NOT NULL,
	CreatedOn datetime NOT NULL,
	CreatedBy varchar(10) NOT NULL,
	LastUpdated datetime NOT NULL,
	AvailableIn TINYTEXT DEFAULT NULL,
	PRIMARY KEY(ProductID)
)ENGING=INNODB;

CREATE TABLE order_header(
	OrderID varchar(50) NOT NULL,
	OrderDate datetime NOT NULL,
	Marketplace varchar(50) NOT NULL,
	TotalAmount FLOAT NOT NULL,
	OrderStatus 
	OrderStatusDesc TINYTEXT NOT NULL,
	OrderRefNum varchar(100) NOT NULL,
	ShippingAgent varchar(50) NOT NULL,
	PRIMARY KEY(OrderID)
)ENGING=INNODB;

CREATE TABLE order_detail(
	OrderID varchar(50) NOT NULL
	OrderDate datetime NOT NULL,
	ProductID varchar(100) NOT NULL,
	SKU varchar(100) NOT NULL,
	ProductName TINYTEXT NOT NULL,
	Quantity int(6) NOT NULL,
	Price FLOAT NOT NULL,
	TotalPrice FLOAT NOT NULL
)ENGING=INNODB;
ALTER TABLE order_detail
  ADD CONSTRAINT FK_OrderID_ordetail FOREIGN KEY (OrderID) REFERENCES order_header(OrderID),
  ADD CONSTRAINT FK_SKU_ordetail FOREIGN KEY (ProductID) REFERENCES product(ProductID);

CREATE TABLE order_status_flg(
	OrderID varchar(50) NOT NULL,
	OrderStatus 
	OrderStatusDesc TINYTEXT NOT NULL,
	OrderStatusDate datetime NOT NULL,
)ENGING=INNODB;
ALTER TABLE order_status_flg
  ADD CONSTRAINT FK_OrderID_orstatflg FOREIGN KEY (OrderID) REFERENCES order_header(OrderID);
